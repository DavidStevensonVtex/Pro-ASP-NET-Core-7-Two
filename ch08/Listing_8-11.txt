@* Listing 8.11 Highlighting category in the Default.cshtml file in the SportsStore/Views/Shared/Components/NavigationMenu folder *@

@model IEnumerable<string>

<div class="d-grid gap-2">
    @{
        string homeClassValue = "btn " + 
            (ViewBag.SelectedCategory == null ? "btn-primary" : "btn-outline-secondary");
    }
    <a class="@homeClassValue"
       asp-action="Index"
       asp-controller="Home"
       asp-route-category="">
        Home
    </a>

    @foreach (string category in Model ?? Enumerable.Empty<string>())
    {
        string classValue = "btn " + (category == ViewBag.SelectedCategory ? "btn-primary" : "btn-outline-secondary");
        <a class="@classValue"
           asp-action="Index"
           asp-controller="Home"
           asp-route-category="@category"
           asp-route-productPage="1">
            @category
        </a>
    }
</div>
